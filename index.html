<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Prison Escape Game</title>

    <style>

        body {

            display: flex;

            justify-content: center;

            align-items: center;

            height: 100vh;

            margin: 0;

            background-color: #222;

        }

        canvas {

            background-color: white;

            border: 2px solid black;

        }

    </style>

</head>

<body>

    <canvas id="gameCanvas"></canvas>

    <script>

        const canvas = document.getElementById("gameCanvas");

        const ctx = canvas.getContext("2d");

        // Set canvas size

        canvas.width = 800;

        canvas.height = 400;

        // Player object

        const player = {

            x: 50,

            y: 300,

            width: 30,

            height: 50,

            speed: 5,

            velocityY: 0,

            gravity: 0.5,

            jumpPower: -10,

            grounded: false

        };

        // Key states

        const keys = {

            right: false,

            left: false,

            up: false

        };

        // Floor

        const ground = {

            x: 0,

            y: 350,

            width: canvas.width,

            height: 50

        };

        // Obstacles

        const obstacles = [

            { x: 300, y: 300, width: 50, height: 50 },

            { x: 500, y: 280, width: 70, height: 70 }

        ];

        // Exit door

        const exit = { x: 750, y: 300, width: 40, height: 50 };

        // Event listeners for movement

        document.addEventListener("keydown", (event) => {

            if (event.key === "ArrowRight") keys.right = true;

            if (event.key === "ArrowLeft") keys.left = true;

            if (event.key === "ArrowUp" && player.grounded) {

                player.velocityY = player.jumpPower;

                player.grounded = false;

            }

        });

        document.addEventListener("keyup", (event) => {

            if (event.key === "ArrowRight") keys.right = false;

            if (event.key === "ArrowLeft") keys.left = false;

        });

        // Collision detection function

        function isColliding(a, b) {

            return (

                a.x < b.x + b.width &&

                a.x + a.width > b.x &&

                a.y < b.y + b.height &&

                a.y + a.height > b.y

            );

        }

        // Game loop

        function update() {

            // Horizontal movement

            if (keys.right) player.x += player.speed;

            if (keys.left) player.x -= player.speed;

            // Apply gravity

            player.velocityY += player.gravity;

            player.y += player.velocityY;

            // Collision with ground

            if (player.y + player.height >= ground.y) {

                player.y = ground.y - player.height;

                player.velocityY = 0;

                player.grounded = true;

            }

            // Collision with obstacles

            obstacles.forEach(obstacle => {

                if (isColliding(player, obstacle)) {

                    if (player.y + player.height - player.velocityY <= obstacle.y) {

                        player.y = obstacle.y - player.height;

                        player.velocityY = 0;

                        player.grounded = true;

                    } else {

                        if (keys.right) player.x -= player.speed;

                        if (keys.left) player.x += player.speed;

                    }

                }

            });

            // Check for escape

            if (isColliding(player, exit)) {

                alert("You escaped!");

                resetGame();

            }

            // Render everything

            draw();

            requestAnimationFrame(update);

        }

        // Drawing function

        function draw() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Draw player

            ctx.fillStyle = "blue";

            ctx.fillRect(player.x, player.y, player.width, player.height);

            // Draw ground

            ctx.fillStyle = "green";

            ctx.fillRect(ground.x, ground.y, ground.width, ground.height);

            // Draw obstacles

            ctx.fillStyle = "red";

            obstacles.forEach(obstacle => {

                ctx.fillRect(obstacle.x, obstacle.y, obstacle.width, obstacle.height);

            });

            // Draw exit door

            ctx.fillStyle = "yellow";

            ctx.fillRect(exit.x, exit.y, exit.width, exit.height);

        }

        // Reset game function

        function resetGame() {

            player.x = 50;

            player.y = 300;

            player.velocityY = 0;

        }

        // Start game

        update();

    </script>

</body>

</html>
